26. d) Dart (reloaded)

    
    
# Dart-Spiel mit 201 Punkten.
# Wenn man überwirft, passiert nichts außer einer Warnung.

print("Spielstart bei 201 Punkten")
punkte = 201

while True:
    wurf = int(input())

    if punkte - wurf < 0:
        print("Überworfen!")
        continue   # Runde wird wiederholt

    punkte -= wurf

    if punkte == 0:
        print("Spiel beendet!")
        break
    else:
        print("Noch", punkte)

30. d) Geometrie

    
    
# Dieses Programm berechnet Umfang und Fläche eines Rechtecks.

def umfang(a, b):
    return 2 * (a + b)

def fläche(a, b):
    return a * b

a = int(input("Seite a:"))
b = int(input("Seite b:"))

U = umfang(a, b)
A = fläche(a, b)

print(f"Das Rechteck mit a={a} und b={b} hat Umfang {U} und Fläche {A}.")
