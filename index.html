punkte = 201

while True:
    wurf = int(input())

    if punkte - wurf < 0:
        print("Überworfen!")
        continue

    punkte -= wurf

    if punkte == 0:
        print("Spiel beendet!")
        break
    else:
        print("Noch", punkte)
